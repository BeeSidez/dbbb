import { useState } from "react"; 
import { useTextSetting, useArraySetting } from "@/lib/editable-settings";
function AccordionItem({ question, answer, isOpen, onToggle }) {
  return (
    <div className="border-b border-border">
      <button
        onClick={onToggle}
        className="flex w-full items-center justify-between py-6 text-left"
      >
        <span className="text-xl md:text-2xl font-medium text-foreground pr-4">
          {question}
        </span>
        <span className="text-4xl font-light text-foreground shrink-0 flex items-center justify-center">
          {isOpen ? "−" : "+"}
        </span>
      </button>
      <div
        className={`grid transition-[grid-template-rows] duration-300 ease-in-out ${
            isOpen ? "grid-rows-[1fr]" : "grid-rows-[0fr]"
        }`}
      >
        <div className="overflow-hidden">
          <p className="text-base sm:text-lg lg:text-xl text-muted-foreground leading-relaxed pb-6">
            {answer}
          </p>
        </div>
      </div>
    </div>
  );
}

export default function Block() {
  const title = useTextSetting({
    name: "title",
    label: "Title",
    initialValue: "FAQs",
    required: true,
  });

  const description = useTextSetting({
    name: "description",
    label: "Description",
    initialValue: "Frequently asked by our users",
  });

  const faqData = useArraySetting({
    name: "faq-data",
    label: "FAQ items",
    schema: {
      question: { type: "text", label: "Question", required: true },
      answer: { type: "longText", label: "Answer", required: true },
    },
    initialValue: [
      {
        question: "Do I need coding skills to use Softr?",
        answer:
          "No, Softr is an AI assisted no-code app builder. Non-technical users can build web apps easily by chatting with AI co-builder or through our visual editor.",
      },
      {
        question: "What can I build with Softr?",
        answer:
          "You can build client portals, internal tools, marketplaces, directories, community platforms, and more — all without code.",
      },
      {
        question: "How does Softr connect to my data?",
        answer:
          "Softr connects directly to Softr Databases, Airtable, Google Sheets, Notion, and other data sources. Changes in your data are reflected in your app in real time.",
      },
      {
        question: "Is Softr suitable for building internal tools?",
        answer:
          "Absolutely. Many teams use Softr to build internal dashboards, admin panels, and workflow tools connected to their existing data.",
      },
      {
        question: "Can I control who sees what in my app?",
        answer:
          "Yes, Softr supports role-based access control. You can create user groups and restrict pages or content based on user roles and permissions.",
      },
      {
        question: "How do I get help if I'm stuck?",
        answer:
          "Softr offers 24/7 live support, and we have a community of experts ready to help!",
      },
    ],
  });

  const [openItems, setOpenItems] = useState(new Set([0]));

  const toggleItem = (index) => {
    setOpenItems((prev) => {
      const next = new Set(prev);
      if (next.has(index)) {
        next.delete(index);
      } else {
        next.add(index);
      }
      return next;
    });
  };

  return (
    <div className="container bg-background py-12 md:py-16 lg:py-20">
      <div className="content">
        <div className="flex flex-col lg:flex-row gap-12 lg:gap-20">
          {/* Left Column - Sticky Header */}
          <div className="lg:w-4/12 lg:sticky lg:top-8 lg:self-start">
          <h2 className="text-4xl md:text-6xl text-foreground mb-3">
            {title}
          </h2>
            {description && (
              <p className="text-2xl md:text-3xl font-medium leading-relaxed text-muted-foreground">
                {description}
              </p>
            )}
          </div>

          {/* Right Column - Custom Accordion */}
          <div className="lg:w-8/12">
            {faqData.map((item, index) => (
              <AccordionItem
                key={index}
                question={item.question}
                answer={item.answer}
                isOpen={openItems.has(index)}
                onToggle={() => toggleItem(index)}
              />
            ))}
          </div>
        </div>
      </div>
    </div>
  );
}
