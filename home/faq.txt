import { useState } from "react"; 
import { useTextSetting, useArraySetting } from "@/lib/editable-settings";
function AccordionItem({ question, answer, isOpen, onToggle }) {
  return (
    <div className="border-b border-border">
      <button
        onClick={onToggle}
        className="flex w-full items-center justify-between py-6 text-left"
      >
        <span className="text-xl md:text-2xl font-medium text-foreground pr-4">
          {question}
        </span>
        <span className="text-4xl font-light text-foreground shrink-0 flex items-center justify-center">
          {isOpen ? "−" : "+"}
        </span>
      </button>
      <div
        className={`grid transition-[grid-template-rows] duration-300 ease-in-out ${
            isOpen ? "grid-rows-[1fr]" : "grid-rows-[0fr]"
        }`}
      >
        <div className="overflow-hidden">
          <p className="text-base sm:text-lg lg:text-xl text-muted-foreground leading-relaxed pb-6">
            {answer}
          </p>
        </div>
      </div>
    </div>
  );
}

export default function Block() {
  const title = useTextSetting({
    name: "title",
    label: "Title",
    initialValue: "FAQs",
    required: true,
  });

  const description = useTextSetting({
    name: "description",
    label: "Description",
    initialValue: "Frequently asked by our users",
  });

  const faqData = useArraySetting({
    name: "faq-data",
    label: "FAQ items",
    schema: {
      question: { type: "text", label: "Question", required: true },
      answer: { type: "longText", label: "Answer", required: true },
    },
    initialValue: [
      {
        question: "is this for friends or dating?",
        answer:
          "friends, always. unboring is a friendship platform, not a dating one. it's currently for women, & it's about meeting people you'd genuinely want in your life. what grows from there is yours.",
      },
      {
        question: "will there be people my age?",
        answer:
          "yes. everyone in your group is within a few years of your age & at a similar life stage, so you're meeting people you can actually relate to.",
      },
      {
        question: "how does matching work?",
        answer:
          "when you join, you take a quick personality test. it places you in one of our five archetypes, & we use those to build balanced groups so every table has the right mix of personalities & the dynamic just works. at the end of each experience, we ask who you'd like to see again. if it's mutual, we set you up on your next experience together.",
      },
      {
        question: "how many people am i meeting?",
        answer:
          "the first portion, usually an intimate shared meal, is with 5 to 7 others in your group. after that, groups from across the city come together for a bigger experience, anywhere from 20 to 100 members.",
      },
      {
        question: "when will my reservation be revealed?",
        answer:
          "you'll receive your exact arrival time & location the day before your experience.",
      },
      {
        question: "when do i find out about the second location?",
        answer:
          "about an hour into the first portion, we'll reveal where you're heading next. half the fun is not knowing until you're already in it.",
      },
      {
        question: "what about dietary restrictions?",
        answer:
          "we ask about dietary needs during onboarding & do our best to accommodate them when we curate your night.",
      },
      {
        question: "what does membership include?",
        answer:
          "we curate your entire night, from your reservation to your group to the experience that follows. food & drinks are down to you (unless stated otherwise).",
      },
      {
        question: "how do i stay connected after?",
        answer:
          "every group gets a chat to keep things going, & the more you show up, host, & follow up with the people you've met, the more connection points & streaks you earn. it rewards the things that actually turn a night out into a friendship.",
      },
    ],
  });

  const [openItems, setOpenItems] = useState(new Set([0]));

  const toggleItem = (index) => {
    setOpenItems((prev) => {
      const next = new Set(prev);
      if (next.has(index)) {
        next.delete(index);
      } else {
        next.add(index);
      }
      return next;
    });
  };

  return (
    <div className="container bg-background py-12 md:py-16 lg:py-20">
      <div className="content">
        <div className="flex flex-col lg:flex-row gap-12 lg:gap-20">
          {/* Left Column - Sticky Header */}
          <div className="lg:w-4/12 lg:sticky lg:top-8 lg:self-start">
          <h2 className="text-4xl md:text-6xl text-foreground mb-3">
            {title}
          </h2>
            {description && (
              <p className="text-2xl md:text-3xl font-medium leading-relaxed text-muted-foreground">
                {description}
              </p>
            )}
          </div>

          {/* Right Column - Custom Accordion */}
          <div className="lg:w-8/12">
            {faqData.map((item, index) => (
              <AccordionItem
                key={index}
                question={item.question}
                answer={item.answer}
                isOpen={openItems.has(index)}
                onToggle={() => toggleItem(index)}
              />
            ))}
          </div>
        </div>
      </div>
    </div>
  );
}
