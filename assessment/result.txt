// pull a clean string out of whatever shape a field/lookup returns (string | {label} | [..])
function val(x) {
  if (x == null) return "";
  if (Array.isArray(x)) return x.map(val).filter(Boolean).join(", ");
  if (typeof x === "object") return x.label || x.value || x.name || x.text || "";
  return String(x);
}

  const [fmt, setFmt] = useState("square");
  const [toast, setToast] = useState("");

  // load html-to-image once (for "save card"), defensively
  useEffect(() => {
    if (typeof window === "undefined" || window.htmlToImage) return;
    const s = document.createElement("script");
    s.src = "https://cdn.jsdelivr.net/npm/html-to-image@1.11.13/dist/html-to-image.js";
    s.async = true;
    document.body.appendChild(s);
  }, []);

  // Softr reads the current detail-page record from the connected datasource.
  const query = useRecord({ select: fields });
  const record = (query && (query.record || query.data || query.row)) || query || null;
  const F = (record && (record.fields || record)) || {};

  const charName = val(F.character);
  const meta = CHAR_META[charName];

  if (!charName || !meta) {
    return (
      <div className="ur" style={{ minHeight: "60vh", display: "grid", placeItems: "center" }}>
        <style>{baseCSS}</style>
        <div style={{ textAlign: "center", fontFamily: "'DM Mono',monospace", color: "#615C70", fontSize: 14 }}>finding your friend character…</div>
      </div>
    );
  }

  const id = meta.id;
  const hue = HUES[id];
  const first = (val(F.name).trim().split(/\s+/)[0]) || "friend";
  const sign = val(F.starSign).toLowerCase();
  const merge = (s) => String(s || "").replace(/\[name\]/g, first);
  const sectionText = (S) => {
    let txt = merge(val(F[S.k]));
    if (S.tint && TINTS[sign]) txt = (txt ? txt + " " : "") + TINTS[sign];
    return txt;
  };

  const origin = (typeof window !== "undefined") ? window.location.origin : "";
  const shareUrl = origin + "/friendship-quiz";
  const shareText = `i'm ${charName} — ${meta.identity}. what kind of friend are you?`;
  const enc = encodeURIComponent;
  const openW = (u) => window.open(u, "_blank", "noopener");
  const flash = (m, ms) => { setToast(m); setTimeout(() => setToast(""), ms || 1800); };
  const copyLink = () => { try { navigator.clipboard.writeText(shareUrl); flash("link copied"); } catch (e) { flash("couldn't copy"); } };
  const INTENTS = {
    whatsapp: () => openW(`https://wa.me/?text=${enc(shareText + " " + shareUrl)}`),

    telegram: () => openW(`https://t.me/share/url?url=${enc(shareUrl)}&text=${enc(shareText)}`),
    reddit: () => openW(`https://www.reddit.com/submit?url=${enc(shareUrl)}&title=${enc(shareText)}`),

  };
  const nativeShare = () => {
    if (navigator.share) navigator.share({ title: "what kind of friend are you?", text: shareText, url: shareUrl }).catch(() => {});

  };
  const onSocial = (k) => {
