// ── scoring (ported from reports.js) ──────────────────────────────
function maxPossible(ids, kind) {
  const max = {}; ids.forEach((id) => (max[id] = 0));
  QUESTIONS.filter((qq) => qq.kind === kind).forEach((qq) => {
    ids.forEach((id) => {
      let best = 0;
      qq.o.forEach((opt) => { if (opt.w && opt.w[id]) best = Math.max(best, opt.w[id]); });
      max[id] += best;
    });
  });
  return max;
}
function scoreQuiz(ans) {
  const cs = {}; CHAR_IDS.forEach((i) => (cs[i] = 0));
  const as = {}; ARCH_IDS.forEach((i) => (as[i] = 0));
  QUESTIONS.forEach((qq) => {
    if (!qq.o || ans[qq.id] == null) return;
    const opt = qq.o[ans[qq.id]];
    if (!opt || !opt.w) return;
    Object.entries(opt.w).forEach(([k, v]) => { if (cs[k] != null) cs[k] += v; if (as[k] != null) as[k] += v; });
  });
  const cMax = maxPossible(CHAR_IDS, "char"), aMax = maxPossible(ARCH_IDS, "type");
  const cN = {}; CHAR_IDS.forEach((i) => (cN[i] = cMax[i] ? +(cs[i] / cMax[i]).toFixed(4) : 0));
  const aN = {}; ARCH_IDS.forEach((i) => (aN[i] = aMax[i] ? +(as[i] / aMax[i]).toFixed(4) : 0));
  const pick = (norm, order, tieId) => {
    const top = Math.max(...Object.values(norm));
    const tied = Object.keys(norm).filter((k) => Math.abs(norm[k] - top) < 1e-9);
    if (tied.length === 1) return tied[0];
    const tq = QUESTIONS.find((x) => x.id === tieId);
    const opt = tq && tq.o[ans[tieId]];
    if (opt && opt.w) {
      const fav = Object.keys(opt.w).sort((a, b) => opt.w[b] - opt.w[a])[0];
      if (tied.includes(fav)) return fav;
    }
    for (const id of order) if (tied.includes(id)) return id;
    return tied[0];
  };
  return { character: pick(cN, CHAR_ORDER, "C4"), archetype: pick(aN, ARCH_ORDER, "A1"), charNorm: cN, archNorm: aN };
}
function starSign(m, d) {
  const v = m * 100 + d;
  if (v >= 321 && v <= 419) return "aries";
  if (v >= 420 && v <= 520) return "taurus";
  if (v >= 521 && v <= 620) return "gemini";
  if (v >= 621 && v <= 722) return "cancer";
  if (v >= 723 && v <= 822) return "leo";
  if (v >= 823 && v <= 922) return "virgo";
  if (v >= 923 && v <= 1022) return "libra";
  if (v >= 1023 && v <= 1121) return "scorpio";
  if (v >= 1122 && v <= 1221) return "sagittarius";
  if (v >= 1222 || v <= 119) return "capricorn";
  if (v >= 120 && v <= 218) return "aquarius";
  return "pisces";
}
